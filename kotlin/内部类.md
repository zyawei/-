
## 成员内部类

```kotlin
/**
 * 代码还未验证
 */
class Outer {
    val name = "Outer"
    inner class Inner {
        fun printParent() {
            // 访问外部类成员时指明类作用域
            println(this@Outer.name)
        }
    }
}
class Test{
    fun main(){
        //创建内部类对象
        val outObj = Outer();
        val innerObj = outObj.Inner()
    }
}
```

## 匿名内部类

#### 接口的匿名内部类

```kotlin
class Outer {
    fun test(inner: Inner) {
    }
    interface Inner {
        fun printParent() 
    }
}
class Test{
    fun main(){
        val out = Outer()
        val inner = out.test(object : Outer.Inner {
            override fun printParent() {
                super.printParent()
            }
        })
    }
}

```



#### 静态类的匿名内部类

```kotlin
class Outer {

    fun test(inner: Inner) {
    }

    open class Inner {
        open fun printParent() {
        }
    }
}
class Test{
    fun main(){
        val out = Outer()
        //1，区别于接口生成的匿名内部类，Inner后面带括号。
		val inner = out.test(object : Outer.Inner())
		//2，与Java相同方法可以重载。
        val inner = out.test(object : Outer.Inner() {
            override fun printParent() {
                super.printParent()
            }
        })
    }
}

```
